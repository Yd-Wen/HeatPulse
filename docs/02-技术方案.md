### Prisma Schema 定义 (prisma/schema.prisma)

```prisma
generator client {
  provider = "prisma-client-js"
}

datasource db {
  provider = "sqlite"
  url      = env("DATABASE_URL")
}

model Keyword {
  id          Int           @id @default(autoincrement())
  keyword     String
  category    String?
  is_active   Boolean       @default(true)
  notify_email String?
  created_at  DateTime      @default(now())
  hotspots    Hotspot[]
}

model Hotspot {
  id               Int          @id @default(autoincrement())
  title            String
  content          String?
  source_url       String
  source_type      String       // search, twitter, api
  keyword_id       Int?
  keyword          Keyword?     @relation(fields: [keyword_id], references: [id])
  relevance_score  Float        // AI 评估的相关性 0-1
  is_fake          Boolean      @default(false)
  ai_summary       String?
  ai_tags          String?      // JSON 数组
  importance       Int?
  published_at     DateTime?
  created_at       DateTime     @default(now())
  notifications    Notification[]
}

model Notification {
  id         Int       @id @default(autoincrement())
  hotspot_id Int
  hotspot    Hotspot   @relation(fields: [hotspot_id], references: [id])
  type       String    // websocket, email
  status     String    // pending, sent, failed
  sent_at    DateTime?
  error      String?
}
```

### Prisma Client 使用

```typescript
// 初始化 Prisma Client
import { PrismaClient } from '@prisma/client';

const prisma = new PrismaClient();

// 插入
const keyword = await prisma.keyword.create({
  data: { keyword: 'AI编程', category: '技术' },

// 查询

  where: { is_active: true },

// 事务

    await tx.keyword.create({ data: kw });
  }
});
```

## WebSocket 实时推送 (ws)

### 服务端实现

```typescript
import { WebSocketServer, WebSocket } from 'ws';

const wss = new WebSocketServer({ port: 3002 });

// 广播给所有客户端
export function broadcast(data: any) {
  const message = JSON.stringify(data);
  wss.clients.forEach((client) => {
    if (client.readyState === WebSocket.OPEN) {
      client.send(message);
    }
  });
}

// 广播新热点
export function broadcastHotspot(hotspot: Hotspot) {
  broadcast({
    type: 'NEW_HOTSPOT',
    data: hotspot,
    timestamp: new Date().toISOString(),
  });
}

wss.on('connection', (ws) => {
  console.log('Client connected');
  
  ws.on('close', () => {
    console.log('Client disconnected');
  });
});
```

### 客户端连接

```typescript
const ws = new WebSocket('ws://localhost:3002');

ws.onmessage = (event) => {
```

```typescript
async function sendHotspotEmail(to: string, hotspot: Hotspot) {

  });
}
```

## 定时任务 (node-cron)

```typescript
import cron from 'node-cron';

// 每 30 分钟执行一次扫描
cron.schedule('*/30 * * * *', async () => {
  console.log('开始热点扫描:', new Date().toISOString());
  await scanHotspots();
});

async function scanHotspots() {
  const keywords = getActiveKeywords();
  
  for (const keyword of keywords) {
    // 多源搜索
    const [webResults, tweets] = await Promise.all([
      searchWeb(keyword.keyword),
      fetchTwitter(keyword.keyword).catch(() => []),
    ]);
    
    // AI 分析
    for (const result of [...webResults, ...tweets]) {
      const analysis = await analyzeWithAI(result);
      
      if (analysis.is_real && analysis.authenticity_score > 70) {
        // 保存热点
        const hotspot = saveHotspot(result, analysis, keyword.id);
        
        // 实时推送
        broadcastHotspot(hotspot);
        
        // 邮件通知
        if (keyword.notify_email) {
          await sendHotspotEmail(keyword.notify_email, hotspot);
        }
      }
    }
  }
}
```
